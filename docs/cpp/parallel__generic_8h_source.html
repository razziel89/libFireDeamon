346 template <typename Tout, typename Tsplit, typename... Tins>
347 GPData<Tout, Tsplit, Tins...>::GPData(
348  bool progress_reports, int nr_subs,
349  std::tuple<std::vector<Tsplit>, std::vector<Tins>...> &input,
350  std::vector<Tout> *output, pthread_mutex_t *mutex, int *progress_bar,
351  int split_factor_in, int split_factor_out, bool interlace) {
352  m_nr_subs = nr_subs > 0 ? nr_subs : 1;
353  m_lengths.reserve(1 + sizeof...(Tins));
354  // If there is only one thread, do not waste time with interlacing and
355  // deinterlacing since that is not necessary
356  m_interlace = m_nr_subs > 1 ? interlace : false;
357  tuple_it::for_each_in_tuple_vector(&input, &m_lengths, get_size_functor());
358  // The following has not yet been filled with values but has to be reserved already
359  m_len_output = output->capacity();
360  m_progress_bar = progress_bar;
361  m_mut = mutex;
362  m_progress_reports = progress_reports;
363  m_split_factor_in = split_factor_in;
364  m_split_factor_out = split_factor_out;
365  // do some sanity checking
366  if (m_len_output % m_split_factor_out != 0) {
367  throw std::invalid_argument("The number of elements in the output data is not "
368  "divisible by the output split factor.");
369  }
370  if (m_lengths[0] % m_split_factor_in != 0) {
371  throw std::invalid_argument("The number of elements in the input data (split "
372  "coloumn) is not divisible by the input split factor.");
373  }
374  if (m_lengths[0] / m_split_factor_in != m_len_output / m_split_factor_out) {
375  throw std::invalid_argument(
376  "The data to be split and the output data do not have the same number of "
377  "elements considering the split factor.");
378  }
379  // allocate output data
380  m_output = (Tout *)malloc(m_len_output * sizeof(Tout));
381  // remember the output vector
382  m_output_vector = output;
383  // copy input data over and allocate beforehand
384  {
385  std::vector<std::tuple<unsigned int, size_t, void *>> sizes_pointers_vec;
386  sizes_pointers_vec.reserve(1 + sizeof...(Tins));
387 
388  tuple_it::for_each_in_tuple_vector(&input, &sizes_pointers_vec,
389  get_size_in_bytes_and_pointer_functor());
390 
391  tuple_it::for_each_in_tuple_vector(
392  &m_data, &sizes_pointers_vec,
393  copy_functor_interlace(m_split_factor_in, m_nr_subs, 0, m_interlace));
394  }
395  // create sub data
396  subdata.reserve(m_nr_subs);
397  const int min_nr_elements_per_sub = (m_len_output / m_split_factor_out) / m_nr_subs;
398  int too_many = (m_len_output / m_split_factor_out) % m_nr_subs;
399  int start = 0;
400  for (int sub = 0; sub < m_nr_subs; ++sub) {
401  const int here_elements = min_nr_elements_per_sub + ((too_many > 0) ? 1 : 0);
402  std::vector<int> here_len_data;
403  here_len_data.reserve(m_lengths.size());
404  // copy assignment
405  typename std::tuple<Tsplit *, Tins *...> here_data = m_data;
406  {
407  int count = 0;
408  std::vector<int>::iterator int_it = m_lengths.begin();
409  // typename std::vector<Tin*>::iterator T_it = m_data.begin();
410  for (; int_it != m_lengths.end(); ++int_it /*, ++T_it*/, ++count) {
411  if (count == 0) {
412  here_len_data.push_back(here_elements);
413  // here_data.push_back((*T_it)+m_split_factor_in*start);
414  // push the vector for the input stream which is being split over
415  // threads forward by the necessary number of steps
416  std::get<0>(here_data) += m_split_factor_in * start;
417  } else {
418  here_len_data.push_back(*int_it);
419  // here_data.push_back(*T_it);
420  }
421  }
422  }
423  subdata.push_back(GPSubData<Tout, Tsplit, Tins...>(
424  // general information about thread
425  progress_reports, sub + 1,
426  // input data
427  here_len_data, here_data,
428  // output data
429  here_elements, m_output + m_split_factor_out * start,
430  // information for parallelization
431  m_mut, m_progress_bar));
432  start += here_elements;
433  too_many -= 1;
434  }
435 }

462 template <typename Tout, typename Tsplit, typename... Tins>
463 void GPData<Tout, Tsplit, Tins...>::TransferOutput(bool empty_check) {
464  if (empty_check && m_output_vector->size() > 0) {
465  throw std::length_error("Output vector should be empty but it is not.");
466  }
467  if (m_interlace) { // deinterlace the output data
468  Tout **temp = (Tout **)malloc(m_nr_subs * sizeof(Tout *));
469  const int min_nr_elements_per_sub = (m_len_output / m_split_factor_out) / m_nr_subs;
470  int too_many = (m_len_output / m_split_factor_out) % m_nr_subs;
471  int start = 0;
472  for (int sub = 0; sub < m_nr_subs; ++sub) {
473  temp[sub] = m_output + (m_split_factor_out * start);
474  const int here_elements = min_nr_elements_per_sub + ((too_many > 0) ? 1 : 0);
475  start += here_elements;
476  too_many -= 1;
477  }
478  for (int i = 0; i < m_len_output / m_split_factor_out; ++i) {
479  int modulus = i % m_nr_subs;
480  for (int j = 0; j < m_split_factor_out; ++j) {
481  m_output_vector->push_back(*(temp[modulus]));
482  ++(temp[modulus]);
483  }
484  }
485  free(temp);
486  } else {
487  Tout *temp = m_output;
488  for (int i = 0; i < m_len_output; ++temp, ++i) {
489  m_output_vector->push_back(*temp);
490  }
491  }
492 }

504 template <typename... Ts>
505 void do_parallel_generic(
509  void *(*thread_func)(void *),
512  PG *globals,
514  bool progress_reports,
518  int nr_calcs,
520  GPData<Ts...> *data) {
521  if (pg_global) {
522  throw std::logic_error("Global data for parallel computation already filled.");
523  } else {
524  pg_global = globals;
525  }
526  signal(SIGINT, signal_callback_handler);
527 
528  for (int i = 0; i < globals->nr_threads; ++i) {
529  int rc = pthread_create(globals->threads + i, NULL, thread_func,
530  (void *)data->GetSubData(i));
531  assert(rc == 0);
532  }
533  char prog_char = '|';
534  if (progress_reports) {
535  bool looping = true;
536  int here_progress_bar = 0;
537  fprintf(stdout, "Progress: %6.2f%% | Total: %d/%d %c", 0.0, 0, nr_calcs, prog_char);
538  fflush(stdout);
539  while (looping) {
540  sleep(1);
541  switch (prog_char) {
542  case '|':
543  prog_char = '/';
544  break;
545  case '/':
546  prog_char = '-';
547  break;
548  case '-':
549  prog_char = '\\';
550  break;
551  case '\\':
552  prog_char = '|';
553  break;
554  }
555  pthread_mutex_lock(&(globals->mutex));
556  here_progress_bar = globals->progress_bar;
557  pthread_mutex_unlock(&(globals->mutex));
558  fprintf(stdout, "%c[2K\r", 27);
559  if (here_progress_bar >= nr_calcs) {
560  fprintf(stdout, "Progress: %6.2f%% | Total: %d/%d ", 100.0, nr_calcs,
561  nr_calcs);
562  looping = false;
563  } else {
564  fprintf(stdout, "Progress: %6.2f%% | Total: %d/%d %c",
565  here_progress_bar * 100.0 / nr_calcs, here_progress_bar, nr_calcs,
566  prog_char);
567  }
568  fflush(stdout);
569  }
570  fprintf(stdout, "\n");
571  fflush(stdout);
572  }
573  for (int i = 0; i < globals->nr_threads; ++i) {
574  pthread_join(*(globals->threads + i), NULL);
575  }
576 }
